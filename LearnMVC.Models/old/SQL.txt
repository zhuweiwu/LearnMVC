2.Insert Use T-SQL:

USE ZhuweiDB;
GO
CREATE PROCEDURE [dbo].[spInsertCustom]
@cname nvarchar(50),
@ename nvarchar(50),
@age int,
@departID int,
@password nvarchar(50)
AS
BEGIN
insert into custom(cname,departID,age,ename,password) values (@cname,@departID,@age,@ename,@password)
END
 
RETURN @@Identity

@departname nvarchar(50),

3.Update T-SQL :

USE ZhuweiDB;
GO
CREATE PROCEDURE [dbo].[spupdatecustom] 
@id int,
@cname nvarchar(50),
@departID int,
@age int,
@ename nvarchar(50),
@password nvarchar(50)
AS
BEGIN
    update 
       custom 
    set
    cname = @cname,
    departID = @departID,
    age = @age,
    ename = @ename,
    password = @password
    where id = @id
END
COMMIT TRAN

GO

(
@departname nvarchar(50),
